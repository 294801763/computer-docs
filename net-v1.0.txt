断线重连(使用循环检测连接状态并重连) - v1.0
--------------
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>
#include <sys/socket.h>
#include <arpa/inet.h>

#define SERVER_IP "127.0.0.1"
#define SERVER_PORT 8080
#define RECONNECT_INTERVAL 5  // 重连间隔时间（秒）

int main() {
    int sockfd;
    struct sockaddr_in server_addr;

    while (1) {
        // 创建socket
        sockfd = socket(AF_INET, SOCK_STREAM, 0);
        if (sockfd < 0) {
            perror("Socket creation failed");
            sleep(RECONNECT_INTERVAL);
            continue;
        }

        // 设置服务器地址
        memset(&server_addr, 0, sizeof(server_addr));
        server_addr.sin_family = AF_INET;
        server_addr.sin_port = htons(SERVER_PORT);
        server_addr.sin_addr.s_addr = inet_addr(SERVER_IP);

        // 尝试连接
        if (connect(sockfd, (struct sockaddr*)&server_addr, sizeof(server_addr)) < 0) {
            perror("Connection failed");
            close(sockfd);
            sleep(RECONNECT_INTERVAL);
            continue;
        }

        printf("Connected to server.\n");

        // 这里可以处理业务逻辑
        // ...

        // 关闭连接
        close(sockfd);
        break;  // 如果需要持续连接，可以去掉break
    }

    return 0;
}

注意:这是最简单的一种方法.通过在主循环中不断尝试连接服务器,如果连接失败,则等待一段时间后再次尝试;简单直接,适合对实时性要求不高的场景.

断线重连(使用心跳包检测连接状态) - v1.1
--------------
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>
#include <sys/socket.h>
#include <arpa/inet.h>

#define SERVER_IP "127.0.0.1"
#define SERVER_PORT 8080
#define HEARTBEAT_INTERVAL 5  // 心跳包发送间隔（秒）
#define TIMEOUT 10  // 超时时间（秒）

int send_heartbeat(int sockfd) {
    char heartbeat_msg[] = "HEARTBEAT";
    if (send(sockfd, heartbeat_msg, strlen(heartbeat_msg), 0) < 0) {
        perror("Heartbeat send failed");
        return -1;
    }
    return 0;
}

int receive_response(int sockfd) {
    char buffer[1024];
    int bytes_received = recv(sockfd, buffer, sizeof(buffer) - 1, 0);
    if (bytes_received <= 0) {
        perror("Receive failed");
        return -1;
    }
    buffer[bytes_received] = '\0';
    printf("Received: %s\n", buffer);
    return 0;
}

int main() {
    int sockfd;
    struct sockaddr_in server_addr;

    while (1) {
        sockfd = socket(AF_INET, SOCK_STREAM, 0);
        if (sockfd < 0) {
            perror("Socket creation failed");
            sleep(5);
            continue;
        }

        memset(&server_addr, 0, sizeof(server_addr));
        server_addr.sin_family = AF_INET;
        server_addr.sin_port = htons(SERVER_PORT);
        server_addr.sin_addr.s_addr = inet_addr(SERVER_IP);

        if (connect(sockfd, (struct sockaddr*)&server_addr, sizeof(server_addr)) < 0) {
            perror("Connection failed");
            close(sockfd);
            sleep(5);
            continue;
        }

        printf("Connected to server.\n");

        while (1) {
            if (send_heartbeat(sockfd) < 0) {
                close(sockfd);
                break;
            }

            if (receive_response(sockfd) < 0) {
                close(sockfd);
                break;
            }

            sleep(HEARTBEAT_INTERVAL);
        }

        printf("Connection lost. Reconnecting...\n");
    }

    return 0;
}

注意:在建立连接后,定时向服务器发送心跳包,以检测连接是否正常.如果在一定时间内没有收到服务器的响应,则认为连接已断开,触发重连;通过心跳包检测连接状态,能够更及时地发现连接异常,适合对实时性要求较高的场景.

断线重连(结合 select 函数检测连接状态) - v1.2
--------------
#include <stdio.h>
#include <stdlib.h>
#include <unistd.h>
#include <string.h>
#include <sys/socket.h>
#include <arpa/inet.h>

#define SERVER_IP "127.0.0.1"
#define SERVER_PORT 8080
#define TIMEOUT 10  // 超时时间（秒）

int check_connection(int sockfd) {
    fd_set readfds;
    struct timeval timeout;

    FD_ZERO(&readfds);
    FD_SET(sockfd, &readfds);

    timeout.tv_sec = TIMEOUT;
    timeout.tv_usec = 0;

    int result = select(sockfd + 1, &readfds, NULL, NULL, &timeout);
    if (result < 0) {
        perror("Select failed");
        return -1;
    } else if (result == 0) {
        printf("Connection timeout.\n");
        return -1;
    } else {
        return 0;
    }
}

int main() {
    int sockfd;
    struct sockaddr_in server_addr;

    while (1) {
        sockfd = socket(AF_INET, SOCK_STREAM, 0);
        if (sockfd < 0) {
            perror("Socket creation failed");
            sleep(5);
            continue;
        }

        memset(&server_addr, 0, sizeof(server_addr));
        server_addr.sin_family = AF_INET;
        server_addr.sin_port = htons(SERVER_PORT);
        server_addr.sin_addr.s_addr = inet_addr(SERVER_IP);

        if (connect(sockfd, (struct sockaddr*)&server_addr, sizeof(server_addr)) < 0) {
            perror("Connection failed");
            close(sockfd);
            sleep(5);
            continue;
        }

        printf("Connected to server.\n");

        while (1) {
            if (check_connection(sockfd) < 0) {
                close(sockfd);
                break;
            }

            // 这里可以处理业务逻辑
            // ...

            sleep(1);
        }

        printf("Connection lost. Reconnecting...\n");
    }

    return 0;
}

注意:使用 select 函数可以检测socket是否可读或可写,从而判断连接是否正常.如果检测到连接异常,则触发重连;select 函数检测连接状态,可以同时处理多个socket,适合复杂的网络通信场景.根据实际需求选择合适的方法即可.
