工程创建和修改步骤
--------------------------
创建新项目->Windows桌面向导->项目名称:MFC->应用程序类型:桌面应用程序.exe
                                         ->其他选项:空项目
MFC项目->属性->高级->字符集:使用多字节字符集
                   ->MFC的使用:在静态库中使用MFC                     #MFC项目 
				   
MFC项目->属性->高级->字符集:使用多字节字符集
                   ->MFC的使用:使用标准 Windows 库                   #WIN32项目

MFC项目->属性->链接器->所有选项->子系统:控制台 (/SUBSYSTEM:CONSOLE)  #控制台程序

MFC项目->属性->链接器->所有选项->子系统:窗口 (/SUBSYSTEM:WINDOWS)    #Window程序
		
注意:vs版本为Vistual Stdio 2022

MFC类向导
--------------------------
菜单->项目->类向导(CTRL + SHIFT + X)

命令:资源控件的命令消息处理

消息:标准windows消息

虚函数:子类重写父类虚函数

成员变量:控件ID与控件变量类型类型绑定(值/控件)

方法:编写成员函数

		PAINTSTRUCT ps = { 0 };

	{
		CString str;

		str.Format("wParam:%d,lParam:%d", wParam, lParam);
		AfxMessageBox(str);
		return TRUE;
	}
};

	{
		CClientDC dc(this);
		dc.Ellipse(300,300,500,500);

	{
		CClientDC dc(this);
		CPen pen(PS_SOLID,2,RGB(255,0,0));
		//CPen* old = dc.SelectObject(&pen);
		//or 
		CPen* old = (CPen*)::SelectObject(dc.m_hDC,pen.m_hObject);
		//dc.Rectangle(150,150,300,300);
		//or 
		::Rectangle(dc.m_hDC,150,150,300,300);
		//dc.SelectObject(old);
		//or 
		::SelectObject(dc.m_hDC,old);
		//pen.DeleteObject();
		//or 
		::DeleteObject(pen.m_hObject);
	}
	
	void OnBrush()
	{
		CClientDC dc(this);
		CBrush brush(RGB(0,255,0));
		CBrush *old = dc.SelectObject(&brush);
		dc.Rectangle(180,180,300,300);
		dc.SelectObject(old);
		brush.DeleteObject();
	}
	
	void OnFont()
	{
		CClientDC dc(this);
		CFont font;
		
		font.CreatePointFont(300,"黑体");
		CFont* old = dc.SelectObject(&font);
		dc.TextOut(200,200,"Font!!!");
		dc.SelectObject(old);
		font.DeleteObject();
	}
	
	void OnBitmap()
	{
		//step.1 添加位图资源
		//step.2 创建一个和当前DC,相匹配的内存DC
		//step.3 将位图的数据发送给内存DC
		//step.4 成像
		//step.5 将位图数据要回来
		//step.6 销毁位图
		//step.7 销毁内存DC

		CClientDC dc(this);
		CDC mem;
		mem.CreateCompatibleDC(&dc); //::CreateCompatibleDC

		CBitmap bmp;
		bmp.LoadBitmap(IDB_ICO);
		CBitmap* old = mem.SelectObject(&bmp); //::LOadBitmap

		dc.BitBlt(210, 210, 48, 48, &mem, 0, 0, SRCCOPY); //::BitBlt

		mem.SelectObject(old); //::SelectObject

		bmp.DeleteObject();  //DeleteObject

		mem.DeleteDC(); //::DeleteDC
	}
};

	{
	
	}
	
	virtual void Serialize(CArchive& ar)
	{
		if (ar.IsStoring())
		{
			ar << m_age << m_score << m_name;
		}
		else
		{
			ar >> m_age >> m_score >> m_name;
		}
	}
	
	int m_age;
	float m_score;
	CString m_name;
};

step.1 添加对话框资源
step.2 顶一个自己的对话框类,管理对话框资源,派生自CDialog或CDialogEx均可

资源文件
  IDC_TEXT    -- 文本框
  IDD_DIALOG1 -- 对话框
**/

#include <afxwin.h>
#include <afxext.h>
#include "resource.h"

class MFCDlg : public CDialog {
	DECLARE_MESSAGE_MAP()
public:
	enum { IDD = IDD_DIALOG1 };
	MFCDlg() : CDialog(IDD)
	{
	
	}
	
	afx_msg void OnDLGOK()
	{
		CDialog::OnOK(); //父类只能把对话框隐藏,并没有销毁.
		//::DestroyWindow(this->m_hWnd); //真正能销毁窗口的接口
	}
	
	afx_msg void OnDLGCANCEL()
	{
		CDialog::OnCancel();
		//::DestroyWindow(this->m_hWnd);
		//this->DestroyWindow();
	}
};

BEGIN_MESSAGE_MAP(MFCDlg, CDialog)
	ON_COMMAND(IDOK,OnDLGOK)
	ON_COMMAND(IDCANCEL,OnDLGCANCEL)
END_MESSAGE_MAP()

class MFCWinApp : public CWinApp {
	DECLARE_MESSAGE_MAP()
public:
	MFCWinApp() noexcept {

	}

	virtual BOOL InitInstance() {
		MFCDlg dlg;   //栈空间
		
		m_pMainWnd = &dlg;
		
		int nRet = dlg.DoModal();
		
		return FALSE; //不再执行MFC库中安排的Run函数
	}
};

BEGIN_MESSAGE_MAP(MFCWinApp, CWinApp)

END_MESSAGE_MAP()

MFCWinApp theApp;

对象和控件绑定
---------------------------
/**
将控件窗口和类对象绑定具有两大作用:
1.如果和数据类对象绑定,对象和控件可以进行数据交换
2.如果和控件类对象绑定,对象就可以代表整个控件

与数据类型对象绑定的使用
----------
1.数据类型对象和控件可实现数据交互
2.重写父类成员虚函数DoDataExchange,在函数内部通过一系列的DDX_xxx函数,实现控件和数据类型对象的数据交互
3.如果需要实现数据交互,调用UpdateData函数
    UpdateData(TRUE)  : 控件->变量
	UpdateData(FALSE) : 变量->控件
	
与控件类型对象绑定的使用
----------
1.控件类型对象和控件可实现对象代表整个控件
2.重写父类成员虚函数DoDataExchange,在函数内部通过一些列的DDX_xxx函数,实现控件句柄和控件类型对象的绑定
3.控件类型对象代表这个控件

资源文件
  IDC_TEXT     -- 文本框
  IDOK         -- 确定
  IDCANCEL     -- 取消
  IDC_KS       -- 按钮: 控件->数据对象
  IDC_SK       -- 按钮: 数据对象->控件
  IDD_DIALOG1  -- 对话框
**/

#include <afxwin.h>
#include <afxext.h>
#include "resource.h"

class MFCDlg : public CDialog {
	DECLARE_MESSAGE_MAP()
public:
	enum { IDD = IDD_DIALOG1 };
	CString m_strText;           //数据类型对象
	
	MFCDlg() : CDialog(IDD)
	{
	
	}
	
	virtual void DoDataExchange(CDataExchange* pDX)   //数据类型对象与控件绑定
	{
		DDX_Text(pDX,IDC_TEXT,m_strText);   //调用一系列的DDX_XXX函数
	}
	
	afx_msg void OnDLGOK()
	{
		CDialog::OnOK(); //父类只能把对话框隐藏,并没有销毁.
		//::DestroyWindow(this->m_hWnd); //真正能销毁窗口的接口
	}
	
	afx_msg void OnDLGCANCEL()
	{
		CDialog::OnCancel();
		//::DestroyWindow(this->m_hWnd);
		//this->DestroyWindow();
	}
	
	afx_msg void OnKS()    //控件->数据对象
	{
		this->UpdateData(TRUE);
		AfxMessageBox(m_strText);
	}
	
	afx_msg void OnSK()   //数据对象->控件
	{
		m_strText = "MFCDlg";
		this->UpdateData(FALSE);
	}
};

BEGIN_MESSAGE_MAP(MFCDlg, CDialog)
	ON_COMMAND(IDOK,OnDLGOK)
	ON_COMMAND(IDCANCEL,OnDLGCANCEL)
	ON_COMMAND(IDC_KS,OnKS)
	ON_COMMAND(IDC_SK,OnSK)
END_MESSAGE_MAP()

class MFCWinApp : public CWinApp {
	DECLARE_MESSAGE_MAP()
public:
	MFCWinApp() noexcept {

	}

	virtual BOOL InitInstance() {
		MFCDlg dlg;   //栈空间
		
		m_pMainWnd = &dlg;
		
		int nRet = dlg.DoModal();
		
		return FALSE; //不再执行MFC库中安排的Run函数
	}
};

BEGIN_MESSAGE_MAP(MFCWinApp, CWinApp)

END_MESSAGE_MAP()

MFCWinApp theApp;

/**
1.控件如何与数据类型对象绑定  -- 做数据交换
2.控件如何与控件类对象绑定    -- 对象代表控件
3.控件的消息如何处理

控件消息的基本原理
----------
WM_COMMAND消息
  LOWORD(wParam)    -  菜单项ID,控件ID
  HIWORD(wParam)    -  菜单项为0
                       控件,通知码(控件发生的事情)
  IParam            -  均无用

资源文件
  IDC_TEXT     -- 文本框
  IDOK         -- 确定
  IDCANCEL     -- 取消
  IDC_KS       -- 按钮: 控件->数据对象
  IDC_SK       -- 按钮: 数据对象->控件
  IDC_CBUTTON  -- 按钮: 练习CButton
  IDC_CEdit    -- 按钮: 练习CEdit
  IDC_CkBox    -- 复选框:练习CheckBox
  IDD_DIALOG1  -- 对话框

下压式按钮
-----------
相关类 -- CButton类:父类为CWnd,封装了关于下压式按钮控件的各种操作
消息   -- 当下压式按钮被单击,WM_COMMAND消息(通知码BN_CLICKED)

编辑框
-----------
相关类    -- CEdit类:父类为CWnd,封装了关于编辑框控件的各种操作
常见操作  -- 获取编辑框文本内容,设置编辑框文本内容
消息      -- 当编辑框文本内容发生变化,WM_COMMAND消息(通知码EN_CHANGE)

复选框
-----------
相关类    -- CButton类:父类为CWnd,封装了关于复选框按钮控件的各种操作
常见操作  -- 获取复选框状态,CButton::GetCheck
          -- 设置复选框状态,CButton::SetCheck
消息      -- 当下压式按钮被单击,WM_COMMAND消息(通知码BN_CLICKED)
**/

#include <afxwin.h>
#include <afxext.h>
#include "resource.h"

class MFCDlg : public CDialog {
	DECLARE_MESSAGE_MAP()
public:
	enum { IDD = IDD_DIALOG1 };
	CString m_strText;           //数据类型对象
	CEdit m_ctrlText;            //文本框 - 控件类型对象
	CButton m_cButton;           //按钮   - 控件类型对象,按钮没有数据交换
	
	CString m_strCEdit;          //数据类型对象
	CEdit m_cEdit;               //编辑框
	
	CButton m_cKbox;             //复选框
	INT    m_varCkbox;
	
	MFCDlg() : CDialog(IDD)
	{
	
	}
	
	BOOL OnInitDialog()           //对话框初始化成员函数
	{
		CDialog::OnInitDialog();
		
		m_cButton.MoveWindow(0,0,180,50);
		m_cButton.SetWindowText("练习CButton控件");
		
		m_cEdit.MoveWindow(301,0,180,30);
		m_cEdit.SetWindowText("练习CEdit");
		
		m_strCEdit = "练习CEdit-数据交换";
		UpdateData(FALSE);   //变量->控件
		
		m_cKbox.SetCheck(TRUE);  //复选框
		
		return TRUE;
	}
	
	virtual void DoDataExchange(CDataExchange* pDX)   //数据类型对象与控件绑定
	{
		CDialog::DoDataExchange(pDX);
		DDX_Text(pDX,IDC_TEXT,m_strText);       //数据类型对象,调用一系列的DDX_XXX函数
		DDX_Control(pDX,IDC_TEXT,m_ctrlText);   //控件类型对象
		
		DDX_Control(pDX,IDC_CBUTTON,m_cButton); //控件类型对象
		
		DDX_Control(pDX,IDC_CEdit,m_cEdit);     //控件类型对象
		DDX_Text(pDX,IDC_CEdit,m_strCEdit);     //数据类型对象
		
		DDX_Control(pDX,IDC_CkBox,m_cKbox);
		DDX_Check(pDX,IDC_CkBox,m_varCkbox);
	}
	
	afx_msg void OnDLGOK()
	{
		m_ctrlText.MoveWindow(0,0,200,200);    //移动文本框
		//CDialog::OnOK(); //父类只能把对话框隐藏,并没有销毁.
		//::DestroyWindow(this->m_hWnd); //真正能销毁窗口的接口
	}
	
	afx_msg void OnDLGCANCEL()
	{
		CDialog::OnCancel();
		//::DestroyWindow(this->m_hWnd);
		//this->DestroyWindow();
	}
	
	afx_msg void OnKS()    //控件->数据对象
	{
		this->UpdateData(TRUE);
		AfxMessageBox(m_strText);
	}
	
	afx_msg void OnSK()   //数据对象->控件
	{
		m_strText = "MFCDlg";
		this->UpdateData(FALSE);
	}
	
	afx_msg void OnDClk()
	{
		AfxMessageBox("Double Clicked Message!!!");
	}
	
	afx_msg void OnCButtonClicked()
	{
		char strEdit[256];
		
		//文本框
		m_cEdit.GetWindowText(strEdit,256);
		
		//复选框
		m_cKbox.SetCheck(!m_cKbox.GetCheck());
		
		AfxMessageBox(strEdit);
	}
	
	afx_msg void OnEditChange()
	{
		//char strEdit[256];
		
		//m_cEdit.GetWindowText(strEdit,256);
		//m_cButton.SetWindowText(strEdit);
		
		//与上面两句效果是一样的.
		UpdateData(TRUE);
		m_cButton.SetWindowText(m_strCEdit);
		//AfxMessageBox("Edit Context Changed Message!!!");
	}
	
	afx_msg void OnCkBoxClk()
	{
		int nState = m_cKbox.GetCheck();
		
		CString sState;
		//sState.Format("%d",nState);
		//or 
		UpdateData(TRUE);
		sState.Format("%d",m_varCkbox);
		
		m_cButton.SetWindowText(sState);
	}
};

BEGIN_MESSAGE_MAP(MFCDlg, CDialog)
	//ON_COMMAND(IDOK,OnDLGOK)
	//or 
	ON_BN_CLICKED(IDOK,OnDLGOK)       //按钮单击,ON_BN_CLICKED按钮专用宏
	//ON_COMMAND(IDCANCEL,OnDLGCANCEL)
	//or 
	ON_BN_CLICKED(IDCANCEL,OnDLGCANCEL)
	//ON_COMMAND(IDC_KS,OnKS)         //or ON_BN_CLICKED(IDC_KS,OnKS),只能处理单击消息.
	//or 
	ON_BN_CLICKED(IDC_KS,OnKS)
	//ON_COMMAND(IDC_SK,OnSK)
	//or 
	ON_BN_CLICKED(IDC_SK,OnSK)
	ON_BN_DOUBLECLICKED(IDC_SK,OnDClk)  //按钮双击,需要修改"控件->数据对象"属性"Notify:True"
	
	ON_BN_CLICKED(IDC_CBUTTON,OnCButtonClicked)       //按钮
	
	ON_EN_CHANGE(IDC_CEdit,OnEditChange)              //文本框
	
	ON_BN_CLICKED(IDC_CkBox,OnCkBoxClk)
END_MESSAGE_MAP()

	}

	{
		m_ctrlText.MoveWindow(0,0,200,200);    //移动文本框

	}

	{
		CDialog::OnCancel();
		//::DestroyWindow(this->m_hWnd);
		//this->DestroyWindow();
	}
	
	afx_msg void OnKS()    //控件->数据对象
	{
		this->UpdateData(TRUE);
		AfxMessageBox(m_strText);
	}
	
	afx_msg void OnSK()   //数据对象->控件
	{
		m_strText = "MFCDlg";
		this->UpdateData(FALSE);
	}
};

{

	DWORD nID = 0;
	
	HANDLE hThread = CreateThread(NULL,0,thread,(LPVOID*)pszText,0,&nID);

/**
原子锁
----------
相关问题 - 多个线程对同一个数据进行原子操作,会产生结果丢失.比如执行++运算时.

使用原子锁函数
-----
LONG InterlockedIncrement(
  LPLONG lpAddend   // variable to increment
);

LONG InterlockedDecrement(
  LPLONG lpAddend   // variable address
);

LONG InterlockedCompareExchange( 
  LPLONG Destination,  // destination address
  LONG Exchange,       // exchange value
  LONG Comperand       // value to compare
);

LONG InterlockedExchange(
  LPLONG Target, // value to exchange
  LONG Value     // new value
);

原子锁的实现 - 直接对数据所在的内存操作,并且在任何一个瞬间只能有一个线程访问.

互斥
----------
相关问题 - 多线程下代码或资源的共享使用

互斥的使用
-------
1.创建互斥
-----
HANDLE CreateMutex(
  LPSECURITY_ATTRIBUTES lpMutexAttributes,  //安全属性
  BOOL bInitialOwner,                       //初始的拥有者TRUE/FALSE
  LPCTSTR lpName                            //线程名称
);创建成功返回互斥句柄

2.等候互斥
-----
WaitFor...    --  互斥的等候遵循谁先等候谁先获取

3.释放互斥
-----
BOOL ReleaseMutex(
  HANDLE hMutex   // handle to mutex
);

4.关闭互斥句柄
-----
BOOL CloseHandle(
  HANDLE hObject   // handle to object
);

事件
----------
相关问题 - 程序之间的通知问题

事件的使用
-------
1.创建事件
-----
HANDLE CreateEvent(
  LPSECURITY_ATTRIBUTES lpEventAttributes, //安全属性
  BOOL bManualReset,                       //事件重置(复位)方式,TRUE手动,FALSE自动
  BOOL bInitialState,                      //事件初始状态,TRUE有信号
  LPCTSTR lpName                           //事件名称
);创建成功返回事件句柄

2.等候事件
-----
WaitForSingleObject / WaitForMultipleObjects

3.触发事件(将事件设置成有信号状态)
-----
BOOL SetEvent(
  HANDLE hEvent   // handle to event
);

4.复位事件
-----
BOOL ResetEvent(
  HANDLE hEvent   // handle to event
);

5.关闭事件
-----
BOOL CloseHandle(
  HANDLE hObject   // handle to object
);

信号量
----------
相关问题 - 类似于事件,解决通知的相关问题;提供一个计数器,可以设置次数

信号量的使用
-------
1.创建信号量
-----
HANDLE CreateSemaphore(
  LPSECURITY_ATTRIBUTES lpSemaphoreAttributes, //安全属性
  LONG lInitialCount,                          //初始化信号量数量
  LONG lMaximumCount,                          //信号量的最大值
  LPCTSTR lpName                               //信号量名称
);创建成功返回信号量句柄

2.等候信号量
-----
WaitForSingleObject / WaitForMultipleObjects   -- 每等候通过一次,信号量的信号减1，直到为0阻塞

3.给信号量指定计数值
-----
BOOL ReleaseSemaphore(
  HANDLE hSemaphore,       //信号量句柄
  LONG lReleaseCount,      //释放数量
  LPLONG lpPreviousCount   //释放前原来信号量的数量,可以为NULL
);

4.关闭事件
-----
BOOL CloseHandle(
  HANDLE hObject   // handle to object
);

HANDLE g_hMutex = 0; //互斥句柄

DWORD CALLBACK thread(LPVOID pParam)
{
	char* pszText = (char*)pParam;
	
	while(1)
	{
		WaitForSingleObject(g_hMutex,INFINITE);

		{
			printf("%c",pszText[i]);
			Sleep(2);
		}
		
		printf("\n");
		ReleaseMutex(g_hMutex);
	}
	
	return 0;
