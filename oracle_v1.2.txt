启动oracle
-------------------
$>sqlplus / as sydba
 >startup open                                  #打开数据库
$>/oracle/app/11.2.0//bin/lsnrctl start         #启动监听服务
$>/oracle/app/11.2.0//bin/lsnrctl stop          #停止监听服务 

手动修改默认密码
-------------------
$>sqlplus / as sydba                                         #首先启动sqlplus.
SQL> alter user sys account unlock;                          #解除用户锁定.
SQL> alter user sys identified by manager;                   #修改密码,然后就可以用用户名：sys 密码：manager 登录了.

$>shutdown <normal|transactional|immediate|abort>                                  #oracle的关闭,已经登录了oracle, normal:等待用户断开已存在的链接,系统发出检查点,然后同步数据文件,控制文件和重做日志文件后关闭 ,transactional：等待客户端结束事务之后,自动断开用户链接,然后再执行normal ,immediate:取消当前所有sql语句,回退所有未完成的事务,然后自动断开,在执行normal ,abort:之中所有sql语句,立即终止实例。

oracle基本操作语句
-------------------
创建表
------------
create table tablename(
  col1 type,
  col2 type
);

创建表（包括主键及外键设置）
------------
create table tablename(
  col1 type constraint pk_tablename primary key,
  col2 type not null,
  col3 type constraint fk_tablename reference tablename(colname),
  col4 type constraint ck_tablename check(col4 in(''xxxx'',''ssss'',...))
);

创建表
------------
create table tablename(
  col1 type primary key,
  col2 type not null,
  col3 type check(col3 in(''xxxx'',''ssss'',....))),
  col4 type reference tablename(colname)
);

$>alter table tablename modify(colname type);                                     #修改列属性
$>alter table tablename add(colname type);                                        #添加列
$>alter table tablename add constraint pk_tablename primary key(colname);         #添加主键约束和非空约束
$>alter table tablename modify(colname not null);
$>alter table tablename drop primary key;                                         #删除主键约束
$>alter table tablename drop constraint pk_tablename                              #删除主键约束
$>alter table tablename disable primary key;                                      #失效约束
$>alter table tablename disable constraint pk_tablename;                          #失效约束
$>alter table tablename enable primary key;                                       #有效约束
$>alter table tablename enable constraint pk_tablename;                           #有效约束
$>alter table tablename drop column colname                                       #删除列
$>alter table tablename set unused(colname);                                      #设置某列不可用,然后删除
$>alter table tablename drop unused columns;                                      #设置某列不可用,然后删除
$>rename tablename1 to tablename2;                                                #修改表名
$>alter tablename1 rename to tablename2;                                          #修改表名
$>truncate table tablename;                                                       #截断表
$>truncate table tablename resue storage;                                         #截断表保留行空间
$>desc table tablename;                                                           #查看表结构
$>drop table tablename;                                                           #删除表
$>insert into tablename values(....);                                             #插入记录
$>insert into tablename(....) values(.....);                                      #插入记录
$>insert into tablename values(&colname1,&colname2,...);                          #带参数对话方式插入行
$>insert into tablename(col1,col2,...) values(val1,val2,.....);                   #插入某几列记录
$>insert into tablename values(col1,null,null,...);                               #为列插入空值
$>select * from tablename;                                                        #查询所有行
$>select col1,col2 from tablename;                                                #查询某几列
$>select distict colname from tablename;                                          #重复行消除
$>select * from tablename where 条件 order by colname;                            #where语句查询
$>update tablename set(colname=?) where colname=?                                 #修改列的数据
$>delete from tablename where 条件                                                #删除行

事务处理
-------------------
$>update tablename set colname(日期)=''30-5月-91'' where 条件;
$>savepoint mark1;
$>delete from tablename where 条件;
$>savepoint mark2;
$>rollback to savepoint mark1;
$>rollback;
