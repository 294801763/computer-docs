uhd(uhd-4.6.0.0.tar.gz) - step.1 
--
$>apt-get install autoconf automake build-essential ccache cmake cpufrequtils doxygen ethtool g++ git inetutils-tools libboost-all-dev libncurses5 libncurses5-dev libusb-1.0-0 libusb-1.0-0-dev libusb-dev python3-dev python3-mako python3-numpy python3-requests python3-scipy python3-setuptools python3-ruamel.yaml 
$>cd <repo>/host/
$>mkdir build && cd build
$>cmake -DLIB_SUFFIX=64 ../
$>cmake -DCMAKE_INSTALL_PREFIX=/opt/uhd ../
$>cmake -DCMAKE_FIND_ROOT_PATH=/usr ../
$>cmake -DENABLE_STATIC_LIBS=ON <...>                   #Static Builds
$>cmake -DNEON_SIMD_ENABLE=OFF <...>                    #ARM platforms
$>cmake -DCMAKE_FIND_ROOT_PATH=/usr -DCMAKE_INSTALL_PREFIX=/usr/ ../
or 
$>cmake -DLIB_SUFFIX=64 -DCMAKE_FIND_ROOT_PATH=/usr -DCMAKE_INSTALL_PREFIX=/usr/ ../
$>make -j`nproc` 
$>make -j`nproc` test 
$>make install 
$>ldconfig
$>mkdir -p /usr/share/uhd/images
$>cp -afv uhd-images_4.6.0.0/* /usr/share/uhd/images
$>uhd_find_devices
$>uhd_usrp_probe
$>uhd_config_info --version

如果版本不兼容情况下使用如下命令烧录镜像

-
$>cd /usr/lib/uhd/utils
$>cp -afv uhd-usrp.rules /etc/udev/rules.d/
$>udevadm control --reload-rules
$>udevadm trigger

防火墙
--
$>apt-get install iptables-persistent                            #安装
$>apt-get purge iptables netfilter-persistent+                   #移除

修改 INPUT 和 FORWARD 链默认规则为DROP
-
iptables -P INPUT DROP            #INPUT链拒绝所有流量进入
iptables -P FORWARD DROP          #FORWARD链拒绝所有流量转发
netfilter-persistent save         #保存进配置文件

修改 INPUT 和 FORWARD 链默认规则为ACCEPT
-
iptables -P INPUT ACCEPT 
iptables -P FORWARD ACCEPT 
netfilter-persistent save
netfilter-persistent reload

放开端口
-
iptables -F
iptables -P INPUT DROP
iptables -P FORWARD DROP 
iptables -A INPUT -p icmp --icmp-type echo-request -j ACCEPT 
iptables -A OUTPUT -p icmp --icmp-type echo-reply -j ACCEPT
iptables -A INPUT -p tcp --dport 22 --jump ACCEPT
iptables-save > /etc/iptables/rules.v4  
netfilter-persistent save 

保存和还原
-
iptables-legacy-save > /etc/iptables/rules.v4    
iptables-legacy-restore -n < /etc/iptables/rules.v4

查看防火墙
-
iptables -L --line-numbers          #查看防火墙
iptables -F                         #清空防火墙
iptables -t filter -L INPUT -v --line-numbers
iptables -t filter -L OUTPUT -v --line-numbers
iptables --table filter --list INPUT --verbose --line-numbers

禁止ICMP
-
iptables -F
iptables -P INPUT ACCEPT 
iptables -P FORWARD ACCEPT 
iptables -A INPUT -p icmp --icmp-type echo-request -j DROP 
iptables -A OUTPUT -p icmp --icmp-type echo-reply -j DROP

放行ICMP
-
iptables -F
iptables -P INPUT ACCEPT 
iptables -P FORWARD ACCEPT 
iptables -D INPUT -p icmp --icmp-type echo-request -j DROP
iptables -D OUTPUT -p icmp --icmp-type echo-reply -j DROP
netfilter-persistent save                   #将规则保存进配置文件

放行ICMP
-
iptables -p icmp -h
iptables -D INPUT -p icmp -j ACCEPT
iptables -D INPUT -p icmp -j DROP -v
iptables -D INPUT -p icmp -j REJECT -v

查看防火墙软件
-
iptables --version
update-alternatives --display iptables
