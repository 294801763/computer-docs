设置互联网时间同步服务器
----
$>w32tm /query /status
$>w32tm /config /manualpeerlist:"time.windows.com" /syncfromflags:manual /reliable:YES /update
$>w32tm /resync
or
$>w32tm /config /manualpeerlist:"0.pool.ntp.org 1.pool.ntp.org" /syncfromflags:manual /reliable:YES /update
$>w32tm /resync

检查硬件时钟
----
$>hwclock --show
$>hwclock --set --date="20241201 12:00:00"
$>hwclock -w
$>reboot

windows配置ntp
----
$>w32tm /query /source
$>w32tm /config /manualpeerlist:"192.168.10.22,192.168.0.1" /syncfromflags:manual /reliable:YES /update
$>net stop w32time
$>net start w32time
$>w32tm /query /source               -- 检查时间同步 
$>w32tm /query /status
$>w32tm /query /peers                -- 列出NTP服务器列表
$>w32tm /resync /rediscover          --  强制同步

NTP客户端
----
$>apt-get install ntp
$>nano /etc/ntp.conf         --  客户端
--
server 192.168.10.22

$>systemctl restart ntp.service
$>systemctl status ntp.service
