程序异常启动调试
-------------------
#include<stdio.h>
#include <stdlib.h>
#include <signal.h>
#include <string.h>
#include <execinfo.h>

void dump(int signo)
{
    char buf[1024];
    char cmd[1024];
    FILE *fh;
	
    snprintf(buf,sizeof(buf),"/proc/%d/cmdline",getpid());
    if(!(fh=fopen(buf,"r")))
	{
		exit(0);
	}
    
    if(!fgets(buf,sizeof(buf),fh))
	{
        exit(0);
    }
	
    fclose(fh);
	
    if(buf[strlen(buf)-1]=='\n')
    {
		buf[strlen(buf)-1]='\0';
	}
	
    snprintf(cmd,sizeof(cmd),"gdb %s %d",buf,getpid());
    system(cmd);
    exit(0);
}

int main(void)
{
    signal(SIGSEGV,&dump);
	
    //...
	
    return 0;
}

异常打印堆栈
-------------------
#include<stdio.h>
#include <stdlib.h>
#include <signal.h>
#include <string.h>
#include <execinfo.h>

void dump(int signo)
{
	size_t i;
	size_t size;
    void *array[10];
    char **strings;
    
    size=backtrace(array,10);
    strings=backtrace_symbols(array,size);

    for(i=0;i<size;i++)
	{
        printf("%s\n",strings[i]);
    }
	
    free(strings);
    exit(0);
}

int main(void)
{
    signal(SIGSEGV,&dump);
	signal(SIGINT,&dump);
	signal(SIGQUIT,&dump);
	
    //...
	
    return 0;
}

注意:编译使用"gcc -rdynameic -no-pie -O0 -gdwarf-3","addr2line -Cfe <program> <address>".
