$>vi x 
----------
ca_passwd=`./bin/openssl rand -hex 1024`
key_passwd=`./bin/openssl rand -hex 1024`
./bin/openssl genrsa -des3 -passout pass:$key_passwd -out demo.key 2048
./bin/openssl rsa -passin pass:$key_passwd -in demo.key -out demo.key
./bin/openssl req -new -key demo.key -out demo.csr
./bin/openssl genrsa -passout pass:$ca_passwd -des3 -out ca.key 2048
./bin/openssl req -passin pass:$ca_passwd -new -x509 -key ca.key -out ca.crt -days 36500
mkdir -pv {certs,crl,req,newcerts,crlnumber}
touch serial index.txt
echo "00" > serial
./bin/openssl ca -passin pass:$ca_passwd -in demo.csr -out demo.crt -cert ca.crt -keyfile ca.key -config ./ssl/openssl.cnf
